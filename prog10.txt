void main()

{

int* wskaznik=NULL;

int i,new;

for(i=0; i<ROZMIAR; i++)

{

printf("podaj nowy element tablicy\n");

dodaj(&wskaznik,i,new);

}

}

void dodaj(int** ptab, int n, int new_elem)

{

int i;

int* nowa_tab=malloc((n+1)*sizeof(int));

	

if (nowa_tab==NULL)

{

free(*ptab);

*ptab=NULL;

return;

} 

else

{

for(i=0; i<n; i++)

nowa_tab[i]=(*ptab)[i];

nowa_tab[n]=new_elem;

free(*ptab);

*ptab=nowa_tab;

}


}
